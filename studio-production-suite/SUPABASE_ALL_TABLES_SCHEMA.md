# Supabase Full App Schema (All Tables)

Paste this entire SQL into **Supabase -> SQL Editor** and run once.

```sql
-- =============================
-- XRKR80HD Studio - Full Schema
-- Idempotent: safe to run again
-- =============================

-- 1) site_profiles
create table if not exists public.site_profiles (
  id bigint generated by default as identity primary key,
  headline text,
  short_bio text,
  full_bio text,
  avatar_url text,
  location text,
  email text,
  instagram_url text,
  youtube_url text,
  spotify_url text,
  soundcloud_url text,
  home_hub_card_image_url text,
  home_legends_card_image_url text,
  home_scene_card_image_url text,
  home_podcast_card_image_url text,
  home_contact_card_image_url text,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

alter table public.site_profiles
  add column if not exists avatar_url text,
  add column if not exists home_hub_card_image_url text,
  add column if not exists home_legends_card_image_url text,
  add column if not exists home_scene_card_image_url text,
  add column if not exists home_podcast_card_image_url text,
  add column if not exists home_contact_card_image_url text;

-- 2) tracks
create table if not exists public.tracks (
  id bigint generated by default as identity primary key,
  title text not null,
  artist_name text not null default 'xrkr80hd',
  genre text,
  description text,
  audio_url text not null,
  cover_image_url text,
  external_url text,
  release_date date,
  is_featured boolean not null default false,
  sort_order integer not null default 0,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

alter table public.tracks
  add column if not exists sort_order integer not null default 0;

create index if not exists tracks_admin_sort_idx
  on public.tracks (sort_order, release_date desc, id desc);

-- 3) bands
create table if not exists public.bands (
  id bigint generated by default as identity primary key,
  name text not null,
  slug text not null,
  era text not null default 'archive' check (era in ('archive', 'scene')),
  years_active text,
  genre text,
  tagline text,
  summary text not null,
  story text,
  image_url text,
  banner_image_url text,
  band_photo_url text,
  is_solo_artist boolean not null default false,
  members_json jsonb,
  is_published boolean not null default true,
  sort_order integer not null default 0,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

alter table public.bands
  add column if not exists banner_image_url text,
  add column if not exists band_photo_url text,
  add column if not exists is_solo_artist boolean not null default false,
  add column if not exists members_json jsonb;

create unique index if not exists bands_slug_uidx on public.bands (slug);
create index if not exists bands_era_pub_sort_idx on public.bands (era, is_published, sort_order);
create index if not exists bands_admin_page_genre_idx on public.bands (era, genre, sort_order, name);

-- 4) podcasts (show-level profiles)
create table if not exists public.podcasts (
  id bigint generated by default as identity primary key,
  title text not null,
  slug text not null,
  hosts text,
  topic text,
  summary text,
  description text,
  cover_image_url text,
  is_published boolean not null default true,
  sort_order integer not null default 0,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

create unique index if not exists podcasts_slug_uidx on public.podcasts (slug);
create index if not exists podcasts_admin_sort_idx on public.podcasts (topic, sort_order, title);

-- 5) podcast_episodes
create table if not exists public.podcast_episodes (
  id bigint generated by default as identity primary key,
  podcast_id bigint references public.podcasts(id) on delete cascade,
  title text not null,
  slug text not null,
  topic text,
  summary text,
  description text,
  audio_url text not null,
  cover_image_url text,
  is_featured boolean not null default false,
  is_published boolean not null default true,
  sort_order integer not null default 0,
  published_at timestamptz,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

alter table public.podcast_episodes
  add column if not exists topic text,
  add column if not exists podcast_id bigint references public.podcasts(id) on delete cascade,
  add column if not exists sort_order integer not null default 0;

create unique index if not exists podcast_episodes_slug_uidx on public.podcast_episodes (slug);
create index if not exists podcast_episodes_pub_date_idx on public.podcast_episodes (is_published, published_at);
create index if not exists podcast_episodes_topic_idx on public.podcast_episodes (topic);
create index if not exists podcast_episodes_podcast_sort_idx on public.podcast_episodes (podcast_id, sort_order, published_at desc, id desc);

-- 6) media_items
create table if not exists public.media_items (
  id bigint generated by default as identity primary key,
  title text not null,
  type text not null check (type in ('photo', 'video', 'music')),
  thumbnail_url text,
  media_url text not null,
  caption text,
  is_featured boolean not null default false,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

create index if not exists media_items_type_idx on public.media_items (type, created_at desc);

-- 7) blog_posts
create table if not exists public.blog_posts (
  id bigint generated by default as identity primary key,
  title text not null,
  slug text not null,
  excerpt text,
  content text not null,
  cover_image_url text,
  published_at timestamptz,
  is_published boolean not null default false,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

create unique index if not exists blog_posts_slug_uidx on public.blog_posts (slug);
create index if not exists blog_posts_pub_date_idx on public.blog_posts (is_published, published_at);

-- 8) projects
create table if not exists public.projects (
  id bigint generated by default as identity primary key,
  title text not null,
  slug text not null,
  summary text not null,
  body text,
  cover_image_url text,
  project_url text,
  repo_url text,
  is_featured boolean not null default false,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

create unique index if not exists projects_slug_uidx on public.projects (slug);

-- 9) admin_users (DB-managed admin accounts)
create table if not exists public.admin_users (
  id bigint generated by default as identity primary key,
  username text not null unique,
  display_name text,
  password_hash text not null,
  is_enabled boolean not null default true,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

create index if not exists admin_users_enabled_idx on public.admin_users (is_enabled);
```
